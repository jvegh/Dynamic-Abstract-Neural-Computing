# This is the main CMakeLists file of the Neuron library
#
# @author János Végh

# Generate the main simulator library

message(HIGHLIGHTED "Configuring ${PROJECT_NAME} SystemC-Qt based 'Neuron' library")

file(GLOB_RECURSE MY_SRCS
    *.cpp
    neurons/*.cpp
)

file(GLOB_RECURSE MY_HEADERS
#    include/*.h
     *.h
     neurons/*.h
     ../src/*.h
)

include_directories(
      ../3rdParty/QSystemC/include
      ../3rdParty/fancytabbar
       ${Qt6Widgets_INCLUDE_DIRS}
#       /opt/systemc/include
    ${SYSTEMC_INCLUDE_DIR}
       )

link_directories(${SystemC_LIBRARY_DIRS})

ADD_LIBRARY(${NEURON_LIB}  STATIC
    ${MY_SRCS}
)

target_link_libraries(${NEURON_LIB}
                      Qt6::Widgets
                      ${SystemC_LIBRARIES}
                      ${Pthread}
 )

set_target_properties(${NEURON_LIB}
                      PROPERTIES OUTPUT_NAME Modules
                      )
target_include_directories(${NEURON_LIB} PUBLIC
  $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
)


INSTALL(FILES  ${MY_SRCS} CMakeLists.txt
        DESTINATION ${NEURON_LIB}
        COMPONENT srcs)

INSTALL(FILES  ${MY_HEADERS}
        DESTINATION ${NEURON_LIB}
        COMPONENT srcs)
