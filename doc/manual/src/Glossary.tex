\newglossaryentry{potential-assisted speed}
{
    name={potential-assisted speed},
    description=\supplementbox{In biological systems, the ions contribute
    an electrostatic component to the forces acting on an ion,
    so the ions increase their speed in the field of other ions.
    Only the component in the direcion of the field gradient
    due to the distribution of freely moving charges changes,
    moderately.}
}

\newglossaryentry{potential-accelerated speed}
{
    name={potential-accelerated speed},
    description=\supplementbox{In biological systems some components
    acquire charges and form surfaces having higher potential,
    which act as an electrostatic accelerator.
    Only the component in the direcion of the field gradient 
    due to charges fixed to those surfaces changes,
    strongly.}
}
